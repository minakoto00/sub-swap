# sub-swap Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Rust CLI/TUI tool that manages multiple `~/.codex/` profiles with AES-256-GCM encryption at rest, OS keychain key storage, and cross-platform support.

**Architecture:** Layered design with pure business logic (crypto, profile store, config) tested independently of OS integrations (keychain, process guard) and presentation (CLI/TUI). Directory paths are injectable via a `Paths` struct so tests use temp directories. The keychain and process guard use trait abstractions for testability.

**Tech Stack:** Rust, clap (CLI), ratatui + crossterm (TUI), aes-gcm (encryption), keyring (OS keychain), sysinfo (process detection), serde/serde_json (serialization), chrono (timestamps), dirs (home directory)

```rust
use std::fmt;
use std::io;

#[derive(Debug)]
pub enum SubSwapError {
    Io(io::Error),
    Json(serde_json::Error),
    Crypto(String),
    Keychain(String),
    ProfileNotFound(String),
    ProfileExists(String),
    ActiveProfile(String),
    NoCodexConfig,
    CodexRunning(Vec<u32>),
    NotInitialized,
}

impl fmt::Display for SubSwapError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            Self::Io(e) => write!(f, "I/O error: {e}"),
            Self::Json(e) => write!(f, "JSON error: {e}"),
            Self::Crypto(msg) => write!(f, "Encryption error: {msg}"),
            Self::Keychain(msg) => write!(f, "Keychain error: {msg}"),
            Self::ProfileNotFound(name) => {
                write!(f, "Profile '{name}' not found. Run `sub-swap list` to see available profiles.")
            }
            Self::ProfileExists(name) => {
                write!(f, "Profile '{name}' already exists. Use a different name or `sub-swap remove {name}` first.")
            }
            Self::ActiveProfile(name) => {
                write!(f, "Cannot remove the active profile '{name}'. Switch to another profile first.")
            }
            Self::NoCodexConfig => write!(f, "No auth.json found in ~/.codex/. Nothing to import."),
            Self::CodexRunning(pids) => {
                let pids_str: Vec<String> = pids.iter().map(|p| p.to_string()).collect();
                write!(
                    f,
                    "Codex is currently running (PID {}). Switching profiles may cause unexpected behavior.",
                    pids_str.join(", ")
                )
            }
            Self::NotInitialized => {
                write!(f, "sub-swap is not initialized. Run `sub-swap` to set up.")
            }
        }
    }
}

impl std::error::Error for SubSwapError {}

impl From<io::Error> for SubSwapError {
    fn from(e: io::Error) -> Self {
        Self::Io(e)
    }
}

impl From<serde_json::Error> for SubSwapError {
    fn from(e: serde_json::Error) -> Self {
        Self::Json(e)
    }
}

pub type Result<T> = std::result::Result<T, SubSwapError>;
```

```rust
// src/config.rs
use serde::{Deserialize, Serialize};
use crate::error::Result;
use crate::paths::Paths;
use std::fs;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AppConfig {
    pub encryption_enabled: bool,
}

impl Default for AppConfig {
    fn default() -> Self {
        Self {
            encryption_enabled: true,
        }
    }
}

impl AppConfig {
    pub fn load(paths: &Paths) -> Result<Self> {
        let path = paths.config_json();
        if !path.exists() {
            return Ok(Self::default());
        }
        let data = fs::read_to_string(&path)?;
        let config: Self = serde_json::from_str(&data)?;
        Ok(config)
    }

    pub fn save(&self, paths: &Paths) -> Result<()> {
        let path = paths.config_json();
        let data = serde_json::to_string_pretty(self)?;
        fs::write(&path, data)?;
        Ok(())
    }
}
```

```rust
// src/profile/mod.rs
pub mod store;
pub mod switch;

use std::collections::BTreeMap;
use serde::{Deserialize, Serialize};
use chrono::Utc;
use crate::error::{SubSwapError, Result};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Profile {
    pub name: String,
    pub notes: Option<String>,
    pub created_at: String,
    pub last_used: Option<String>,
}

impl Profile {
    pub fn new(name: &str, notes: Option<String>) -> Self {
        Self {
            name: name.to_string(),
            notes,
            created_at: Utc::now().to_rfc3339(),
            last_used: None,
        }
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ProfileIndex {
    pub version: u32,
    pub active_profile: Option<String>,
    pub profiles: BTreeMap<String, Profile>,
}

impl ProfileIndex {
    pub fn new() -> Self {
        Self {
            version: 1,
            active_profile: None,
            profiles: BTreeMap::new(),
        }
    }

    pub fn add(&mut self, profile: Profile) {
        self.profiles.insert(profile.name.clone(), profile);
    }

    pub fn get(&self, name: &str) -> Option<&Profile> {
        self.profiles.get(name)
    }

    pub fn remove(&mut self, name: &str) -> Result<Profile> {
        if self.active_profile.as_deref() == Some(name) {
            return Err(SubSwapError::ActiveProfile(name.to_string()));
        }
        self.profiles
            .remove(name)
            .ok_or_else(|| SubSwapError::ProfileNotFound(name.to_string()))
    }

    pub fn rename(&mut self, old: &str, new: &str) -> Result<()> {
        if self.profiles.contains_key(new) {
            return Err(SubSwapError::ProfileExists(new.to_string()));
        }
        let mut profile = self
            .profiles
            .remove(old)
            .ok_or_else(|| SubSwapError::ProfileNotFound(old.to_string()))?;
        profile.name = new.to_string();
        if self.active_profile.as_deref() == Some(old) {
            self.active_profile = Some(new.to_string());
        }
        self.profiles.insert(new.to_string(), profile);
        Ok(())
    }

    pub fn set_active(&mut self, name: &str) {
        self.active_profile = Some(name.to_string());
        if let Some(profile) = self.profiles.get_mut(name) {
            profile.last_used = Some(Utc::now().to_rfc3339());
        }
    }

    pub fn set_note(&mut self, name: &str, note: &str) -> Result<()> {
        let profile = self
            .profiles
            .get_mut(name)
            .ok_or_else(|| SubSwapError::ProfileNotFound(name.to_string()))?;
        profile.notes = Some(note.to_string());
        Ok(())
    }

    pub fn names(&self) -> Vec<&str> {
        self.profiles.keys().map(|s| s.as_str()).collect()
    }
}
```

```rust
impl ProfileStore {
    pub fn init(paths: &Paths) -> Result<Self> {
        let index = ProfileIndex::new();
        let store = Self { index };
        store.save(paths)?;
        Ok(store)
    }

    pub fn load(paths: &Paths) -> Result<Self> {
        let data = fs::read_to_string(paths.profiles_json())?;
        let index: ProfileIndex = serde_json::from_str(&data)?;
        Ok(Self { index })
    }

    pub fn load_or_init(paths: &Paths) -> Result<Self> {
        if paths.profiles_json().exists() {
            Self::load(paths)
        } else {
            Self::init(paths)
        }
    }

    pub fn save(&self, paths: &Paths) -> Result<()> {
        let data = serde_json::to_string_pretty(&self.index)?;
        fs::write(paths.profiles_json(), data)?;
        Ok(())
    }

    pub fn save_profile_files(
        paths: &Paths,
        name: &str,
        auth_data: &[u8],
        config_data: &[u8],
        encrypted: bool,
    ) -> Result<()> {
        let dir = paths.profile_dir(name);
        fs::create_dir_all(&dir)?;

        if encrypted {
            fs::write(dir.join("auth.json.enc"), auth_data)?;
            fs::write(dir.join("config.toml.enc"), config_data)?;
            // Remove plaintext versions if they exist
            let _ = fs::remove_file(dir.join("auth.json"));
            let _ = fs::remove_file(dir.join("config.toml"));
        } else {
            fs::write(dir.join("auth.json"), auth_data)?;
            fs::write(dir.join("config.toml"), config_data)?;
            // Remove encrypted versions if they exist
            let _ = fs::remove_file(dir.join("auth.json.enc"));
            let _ = fs::remove_file(dir.join("config.toml.enc"));
        }

        #[cfg(unix)]
        {
            use std::os::unix::fs::PermissionsExt;
            let perms = fs::Permissions::from_mode(0o600);
            for entry in fs::read_dir(&dir)? {
                let entry = entry?;
                fs::set_permissions(entry.path(), perms.clone())?;
            }
        }

        Ok(())
    }

    pub fn load_profile_files(paths: &Paths, name: &str) -> Result<(Vec<u8>, Vec<u8>)> {
        let dir = paths.profile_dir(name);

        let auth = if dir.join("auth.json.enc").exists() {
            fs::read(dir.join("auth.json.enc"))?
        } else {
            fs::read(dir.join("auth.json"))?
        };

        let config = if dir.join("config.toml.enc").exists() {
            fs::read(dir.join("config.toml.enc"))?
        } else {
            fs::read(dir.join("config.toml"))?
        };

        Ok((auth, config))
    }

    pub fn profile_is_encrypted(paths: &Paths, name: &str) -> bool {
        let dir = paths.profile_dir(name);
        dir.join("auth.json.enc").exists()
    }

    pub fn delete_profile_dir(paths: &Paths, name: &str) -> Result<()> {
        let dir = paths.profile_dir(name);
        if dir.exists() {
            fs::remove_dir_all(&dir)?;
        }
        Ok(())
    }

    pub fn rename_profile_dir(paths: &Paths, old: &str, new: &str) -> Result<()> {
        let old_dir = paths.profile_dir(old);
        let new_dir = paths.profile_dir(new);
        fs::rename(&old_dir, &new_dir)?;
        Ok(())
    }
}
```

```rust
use aes_gcm::aead::{Aead, KeyInit};
use aes_gcm::{Aes256Gcm, Key, Nonce};
use rand::RngCore;

use crate::error::{SubSwapError, Result};

pub fn generate_key() -> [u8; 32] {
    let mut key = [0u8; 32];
    rand::rng().fill_bytes(&mut key);
    key
}

pub fn encrypt(plaintext: &[u8], key: &[u8; 32]) -> Result<Vec<u8>> {
    let cipher = Aes256Gcm::new(Key::<Aes256Gcm>::from_slice(key));
    let mut nonce_bytes = [0u8; 12];
    rand::rng().fill_bytes(&mut nonce_bytes);
    let nonce = Nonce::from_slice(&nonce_bytes);

    let ciphertext = cipher
        .encrypt(nonce, plaintext)
        .map_err(|e| SubSwapError::Crypto(e.to_string()))?;

    let mut output = Vec::with_capacity(12 + ciphertext.len());
    output.extend_from_slice(&nonce_bytes);
    output.extend_from_slice(&ciphertext);
    Ok(output)
}

pub fn decrypt(data: &[u8], key: &[u8; 32]) -> Result<Vec<u8>> {
    if data.len() < 28 {
        // Minimum: 12 nonce + 0 plaintext + 16 tag
        return Err(SubSwapError::Crypto("Data too short to be valid ciphertext".into()));
    }
    let (nonce_bytes, ciphertext) = data.split_at(12);
    let cipher = Aes256Gcm::new(Key::<Aes256Gcm>::from_slice(key));
    let nonce = Nonce::from_slice(nonce_bytes);

    let plaintext = cipher
        .decrypt(nonce, ciphertext)
        .map_err(|e| SubSwapError::Crypto(e.to_string()))?;
    Ok(plaintext)
}
```

```rust
use crate::error::{SubSwapError, Result};

pub trait KeyStore {
    fn get_key(&self) -> Result<[u8; 32]>;
    fn set_key(&self, key: &[u8; 32]) -> Result<()>;
    fn has_key(&self) -> bool;
}

// Production implementation using OS keychain
pub struct OsKeyStore;

impl OsKeyStore {
    const SERVICE: &'static str = "sub-swap";
    const ACCOUNT: &'static str = "encryption-key";

    pub fn new() -> Self {
        Self
    }
}

impl KeyStore for OsKeyStore {
    fn get_key(&self) -> Result<[u8; 32]> {
        let entry = keyring::Entry::new(Self::SERVICE, Self::ACCOUNT)
            .map_err(|e| SubSwapError::Keychain(e.to_string()))?;
        let secret = entry
            .get_secret()
            .map_err(|e| SubSwapError::Keychain(e.to_string()))?;
        if secret.len() != 32 {
            return Err(SubSwapError::Keychain("Invalid key length in keychain".into()));
        }
        let mut key = [0u8; 32];
        key.copy_from_slice(&secret);
        Ok(key)
    }

    fn set_key(&self, key: &[u8; 32]) -> Result<()> {
        let entry = keyring::Entry::new(Self::SERVICE, Self::ACCOUNT)
            .map_err(|e| SubSwapError::Keychain(e.to_string()))?;
        entry
            .set_secret(key)
            .map_err(|e| SubSwapError::Keychain(e.to_string()))?;
        Ok(())
    }

    fn has_key(&self) -> bool {
        let entry = match keyring::Entry::new(Self::SERVICE, Self::ACCOUNT) {
            Ok(e) => e,
            Err(_) => return false,
        };
        entry.get_secret().is_ok()
    }
}
```

```rust
use crate::error::{SubSwapError, Result};

pub trait CodexGuard {
    fn check(&self) -> Result<()>;
    fn find_codex_pids(&self) -> Vec<u32>;
}

pub struct OsGuard;

impl OsGuard {
    pub fn new() -> Self {
        Self
    }
}

impl CodexGuard for OsGuard {
    fn find_codex_pids(&self) -> Vec<u32> {
        use sysinfo::System;
        let mut sys = System::new();
        sys.refresh_processes(sysinfo::ProcessesToUpdate::All, true);

        sys.processes()
            .values()
            .filter(|p| {
                let name = p.name().to_string_lossy().to_lowercase();
                name == "codex" || name == "codex.exe"
            })
            .map(|p| p.pid().as_u32())
            .collect()
    }

    fn check(&self) -> Result<()> {
        let pids = self.find_codex_pids();
        if pids.is_empty() {
            Ok(())
        } else {
            Err(SubSwapError::CodexRunning(pids))
        }
    }
}

        self.pids.clone()
    }

    fn check(&self) -> Result<()> {
        if self.pids.is_empty() {
            Ok(())
        } else {
            Err(SubSwapError::CodexRunning(self.pids.clone()))
        }
    }
}
```

```rust
use std::fs;
use crate::crypto;
use crate::error::{SubSwapError, Result};
use crate::paths::Paths;
use crate::profile::Profile;
use crate::profile::store::ProfileStore;

pub fn switch_profile(
    paths: &Paths,
    target: &str,
    key: &[u8; 32],
    encrypt: bool,
) -> Result<()> {
    let mut store = ProfileStore::load(paths)?;

    // No-op if already active
    if store.index.active_profile.as_deref() == Some(target) {
        return Ok(());
    }

    // Verify target exists
    if store.index.get(target).is_none() {
        return Err(SubSwapError::ProfileNotFound(target.to_string()));
    }

    // 1. Load target profile files (may be encrypted)
    let (target_auth, target_config) = ProfileStore::load_profile_files(paths, target)?;

    let (target_auth_plain, target_config_plain) =
        if ProfileStore::profile_is_encrypted(paths, target) {
            (crypto::decrypt(&target_auth, key)?, crypto::decrypt(&target_config, key)?)
        } else {
            (target_auth, target_config)
        };

    // 2. Read current codex files (the active profile's live data)
    let current_auth = fs::read(paths.codex_auth())?;
    let current_config = fs::read(paths.codex_config())?;

    // 3. Save current active profile back to storage
    if let Some(active_name) = store.index.active_profile.clone() {
        let (save_auth, save_config) = if encrypt {
            (crypto::encrypt(&current_auth, key)?, crypto::encrypt(&current_config, key)?)
        } else {
            (current_auth, current_config)
        };
        ProfileStore::save_profile_files(paths, &active_name, &save_auth, &save_config, encrypt)?;
    }

    // 4. Write target profile to codex directory
    fs::write(paths.codex_auth(), &target_auth_plain)?;
    fs::write(paths.codex_config(), &target_config_plain)?;

    #[cfg(unix)]
    {
        use std::os::unix::fs::PermissionsExt;
        let perms = fs::Permissions::from_mode(0o600);
        fs::set_permissions(paths.codex_auth(), perms.clone())?;
        fs::set_permissions(paths.codex_config(), perms)?;
    }

    // 5. Update index
    store.index.set_active(target);
    store.save(paths)?;

    Ok(())
}

pub fn add_profile_from_codex(
    paths: &Paths,
    store: &mut ProfileStore,
    name: &str,
    notes: Option<&str>,
    key: &[u8; 32],
    encrypt: bool,
) -> Result<()> {
    if store.index.get(name).is_some() {
        return Err(SubSwapError::ProfileExists(name.to_string()));
    }

    if !paths.codex_auth().exists() {
        return Err(SubSwapError::NoCodexConfig);
    }

    let auth = fs::read(paths.codex_auth())?;
    let config = fs::read(paths.codex_config())?;

    let (save_auth, save_config) = if encrypt {
        (crypto::encrypt(&auth, key)?, crypto::encrypt(&config, key)?)
    } else {
        (auth, config)
    };

    ProfileStore::save_profile_files(paths, name, &save_auth, &save_config, encrypt)?;

    let profile = Profile::new(name, notes.map(String::from));
    store.index.add(profile);
    store.index.set_active(name);
    store.save(paths)?;

    Ok(())
}

pub fn add_profile_from_path(
    paths: &Paths,
    store: &mut ProfileStore,
    name: &str,
    source: &std::path::Path,
    notes: Option<&str>,
    key: &[u8; 32],
    encrypt: bool,
) -> Result<()> {
    if store.index.get(name).is_some() {
        return Err(SubSwapError::ProfileExists(name.to_string()));
    }

    let auth_path = source.join("auth.json");
    if !auth_path.exists() {
        return Err(SubSwapError::NoCodexConfig);
    }

    let auth = fs::read(auth_path)?;
    let config = fs::read(source.join("config.toml"))?;

    let (save_auth, save_config) = if encrypt {
        (crypto::encrypt(&auth, key)?, crypto::encrypt(&config, key)?)
    } else {
        (auth, config)
    };

    ProfileStore::save_profile_files(paths, name, &save_auth, &save_config, encrypt)?;

    let profile = Profile::new(name, notes.map(String::from));
    store.index.add(profile);
    store.save(paths)?;

    Ok(())
}

pub fn decrypt_profile_to_stdout(
    paths: &Paths,
    name: &str,
    key: &[u8; 32],
) -> Result<(String, String)> {
    let store = ProfileStore::load(paths)?;
    if store.index.get(name).is_none() {
        return Err(SubSwapError::ProfileNotFound(name.to_string()));
    }

    let (auth_data, config_data) = ProfileStore::load_profile_files(paths, name)?;

    let (auth_plain, config_plain) = if ProfileStore::profile_is_encrypted(paths, name) {
        (crypto::decrypt(&auth_data, key)?, crypto::decrypt(&config_data, key)?)
    } else {
        (auth_data, config_data)
    };

    let auth_str = String::from_utf8_lossy(&auth_plain).to_string();
    let config_str = String::from_utf8_lossy(&config_plain).to_string();

    Ok((auth_str, config_str))
}
```

```rust
use crate::config::AppConfig;
use crate::crypto;
use crate::crypto::keychain::{KeyStore, OsKeyStore};

use crate::profile::switch;
use crate::{Cli, Commands, ConfigAction};

pub fn run(cli: Cli) -> Result<()> {
    let paths = Paths::new();

    match cli.command {
        None => {
            // Launch TUI or first-launch wizard
            if !paths.profiles_json().exists() {
                crate::tui::wizard::run_first_launch(&paths)?;
            } else {
                crate::tui::run_tui(&paths)?;
            }
            Ok(())
        }
        Some(cmd) => {
            // All subcommands require initialization (except we trigger first launch if needed)
            if !paths.profiles_json().exists() {
                crate::tui::wizard::run_first_launch(&paths)?;
            }
            run_command(cmd, &paths)
        }
    }
}

fn run_command(cmd: Commands, paths: &Paths) -> Result<()> {
    match cmd {
        Commands::List { verbose } => cmd_list(paths, verbose),
        Commands::Use { name, force } => cmd_use(paths, &name, force),
        Commands::Add { name, from, note } => cmd_add(paths, &name, from.as_deref(), note.as_deref()),
        Commands::Remove { name } => cmd_remove(paths, &name),
        Commands::Rename { old, new } => cmd_rename(paths, &old, &new),
        Commands::Note { name, text } => cmd_note(paths, &name, &text),
        Commands::Decrypt { name } => cmd_decrypt(paths, &name),
        Commands::Config { action } => cmd_config(paths, action),
    }
}

fn cmd_list(paths: &Paths, verbose: bool) -> Result<()> {
    let store = ProfileStore::load(paths)?;
    let active = store.index.active_profile.as_deref();

    for name in store.index.names() {
        let profile = store.index.get(name).unwrap();
        let marker = if active == Some(name) { "*" } else { " " };
        let notes = profile.notes.as_deref().unwrap_or("");

        if verbose {
            let last_used = profile.last_used.as_deref().unwrap_or("never");
            let suffix = if active == Some(name) { " (active)" } else { "" };
            println!("{marker} {name:<16} {notes:<40} Last used: {last_used}{suffix}");
        } else {
            println!("{marker} {name:<16} {notes}");
        }
    }
    Ok(())
}

fn cmd_use(paths: &Paths, name: &str, force: bool) -> Result<()> {
    let store = ProfileStore::load(paths)?;
    if store.index.active_profile.as_deref() == Some(name) {
        println!("Profile '{name}' is already active.");
        return Ok(());
    }

    if !force {
        let guard = OsGuard::new();
        if let Err(e) = guard.check() {
            eprintln!("{e}");
            eprintln!("Use --force to switch anyway.");
            return Err(e);
        }
    }

    let config = AppConfig::load(paths)?;
    let keystore = OsKeyStore::new();
    let key = get_or_warn_key(&keystore, config.encryption_enabled)?;

    switch::switch_profile(paths, name, &key, config.encryption_enabled)?;
    println!("Switched to profile '{name}'.");
    Ok(())
}

fn cmd_add(paths: &Paths, name: &str, from: Option<&str>, note: Option<&str>) -> Result<()> {
    let mut store = ProfileStore::load(paths)?;
    let config = AppConfig::load(paths)?;
    let keystore = OsKeyStore::new();
    let key = get_or_warn_key(&keystore, config.encryption_enabled)?;

    match from {
        Some(dir) => {
            switch::add_profile_from_path(
                paths, &mut store, name, Path::new(dir), note, &key, config.encryption_enabled,
            )?;
            println!("Profile '{name}' added from {dir}.");
        }
        None => {
            switch::add_profile_from_codex(
                paths, &mut store, name, note, &key, config.encryption_enabled,
            )?;
            println!("Profile '{name}' added from ~/.codex/.");
        }
    }
    Ok(())
}

fn cmd_remove(paths: &Paths, name: &str) -> Result<()> {
    let mut store = ProfileStore::load(paths)?;
    store.index.remove(name)?;
    ProfileStore::delete_profile_dir(paths, name)?;
    store.save(paths)?;
    println!("Profile '{name}' removed.");
    Ok(())
}

fn cmd_rename(paths: &Paths, old: &str, new: &str) -> Result<()> {
    let mut store = ProfileStore::load(paths)?;
    store.index.rename(old, new)?;
    ProfileStore::rename_profile_dir(paths, old, new)?;
    store.save(paths)?;
    println!("Profile '{old}' renamed to '{new}'.");

}

    store.index.set_note(name, text)?;
    store.save(paths)?;

}

fn cmd_decrypt(paths: &Paths, name: &str) -> Result<()> {
    let config = AppConfig::load(paths)?;
    let keystore = OsKeyStore::new();
    let key = get_or_warn_key(&keystore, config.encryption_enabled)?;

    let (auth, config_toml) = switch::decrypt_profile_to_stdout(paths, name, &key)?;

    println!("=== auth.json ===");
    println!("{auth}");
    println!("=== config.toml ===");
    println!("{config_toml}");
    Ok(())
}

fn cmd_config(paths: &Paths, action: ConfigAction) -> Result<()> {
    match action {
        ConfigAction::Show => {
            let config = AppConfig::load(paths)?;
            println!("encryption_enabled: {}", config.encryption_enabled);
            Ok(())
        }
        ConfigAction::Set { key, value } => {
            let mut config = AppConfig::load(paths)?;
            match key.as_str() {
                "encryption" => {
                    let enabled = match value.as_str() {
                        "true" => true,
                        "false" => false,
                        _ => {
                            eprintln!("Invalid value for encryption: use 'true' or 'false'");
                            return Ok(());
                        }
                    };
                    let was_enabled = config.encryption_enabled;
                    config.encryption_enabled = enabled;
                    config.save(paths)?;

                    // Re-encrypt or decrypt all non-active profiles
                    if was_enabled != enabled {
                        let keystore = OsKeyStore::new();
                        let key = get_or_warn_key(&keystore, true)?;
                        toggle_all_profiles(paths, &key, enabled)?;
                    }

                    println!("encryption set to {enabled}.");
                }
                _ => {
                    eprintln!("Unknown config key: {key}. Available keys: encryption");
                }
            }
            Ok(())
        }
    }
}

fn get_or_warn_key(keystore: &impl KeyStore, needed: bool) -> Result<[u8; 32]> {
    if !needed {
        // Return a dummy key — won't be used for actual encryption
        return Ok([0u8; 32]);
    }
    keystore.get_key()
}

fn toggle_all_profiles(paths: &Paths, key: &[u8; 32], encrypt: bool) -> Result<()> {
    let store = ProfileStore::load(paths)?;
    let active = store.index.active_profile.as_deref();

    for name in store.index.names() {
        if active == Some(name) {
            continue; // Skip active profile
        }

        let (auth, config_data) = ProfileStore::load_profile_files(paths, name)?;
        let is_enc = ProfileStore::profile_is_encrypted(paths, name);

        if encrypt && !is_enc {
            let enc_auth = crypto::encrypt(&auth, key)?;
            let enc_config = crypto::encrypt(&config_data, key)?;
            ProfileStore::save_profile_files(paths, name, &enc_auth, &enc_config, true)?;
        } else if !encrypt && is_enc {
            let dec_auth = crypto::decrypt(&auth, key)?;
            let dec_config = crypto::decrypt(&config_data, key)?;
            ProfileStore::save_profile_files(paths, name, &dec_auth, &dec_config, false)?;
        }
    }
    Ok(())
}
```

```rust
fn setup_encryption(paths: &Paths) -> Result<([u8; 32], bool)> {
    println!();
    println!("sub-swap encrypts inactive profiles by default using a key");
    println!("stored in your OS keychain.");
    let enable = prompt_yn("Generate encryption key now?", true)?;

    if enable {
        let keystore = OsKeyStore::new();
        let key = crypto::generate_key();
        keystore.set_key(&key)?;

        let config = AppConfig { encryption_enabled: true };
        config.save(paths)?;

        #[cfg(target_os = "macos")]
        println!("  Encryption key stored in macOS Keychain.");
        #[cfg(target_os = "linux")]
        println!("  Encryption key stored in system keyring.");
        #[cfg(target_os = "windows")]
        println!("  Encryption key stored in Windows Credential Manager.");

        Ok((key, true))
    } else {
        let config = AppConfig { encryption_enabled: false };
        config.save(paths)?;
        println!("  Encryption disabled. Enable later with `sub-swap config set encryption true`.");

fn prompt_yn(question: &str, default_yes: bool) -> Result<bool> {
    let hint = if default_yes { "[Y/n]" } else { "[y/N]" };
    print!("{question} {hint} ");
    io::stdout().flush()?;

    let mut input = String::new();
    io::stdin().lock().read_line(&mut input)?;
    let trimmed = input.trim().to_lowercase();

    if trimmed.is_empty() {
        return Ok(default_yes);
    }
    Ok(trimmed.starts_with('y'))
}

fn prompt_string(question: &str, default: Option<&str>) -> Result<String> {
    match default {
        Some(d) => print!("{question} [{d}]: "),
        None => print!("{question}: "),
    }
    io::stdout().flush()?;

    let mut input = String::new();
    io::stdin().lock().read_line(&mut input)?;
    let trimmed = input.trim().to_string();

    if trimmed.is_empty() {
        match default {
            Some(d) => Ok(d.to_string()),
            None => Ok(trimmed),
        }
    } else {
        Ok(trimmed)
    }
}

fn prompt_string_optional(question: &str) -> Result<Option<String>> {
    print!("{question}: ");
    io::stdout().flush()?;

    let mut input = String::new();
    io::stdin().lock().read_line(&mut input)?;
    let trimmed = input.trim().to_string();

    if trimmed.is_empty() {
        Ok(None)
    } else {
        Ok(Some(trimmed))
    }
}
```

```rust
    pub selected: usize,
    pub profile_names: Vec<String>,
    pub active_profile: Option<String>,

    pub input_buffer: String,
    pub message: Option<String>,

}

impl AppState {
    pub fn from_index(index: &ProfileIndex) -> Self {
        let profile_names: Vec<String> = index.names().iter().map(|s| s.to_string()).collect();
        Self {
            screen: AppScreen::Main,
            selected: 0,
            profile_names,
            active_profile: index.active_profile.clone(),
            pending_action: None,
            input_buffer: String::new(),
            message: None,
            decrypt_output: None,
            should_quit: false,
        }
    }

        if self.selected > 0 {
            self.selected -= 1;
        }
    }

        if self.selected + 1 < self.profile_names.len() {
            self.selected += 1;
        }
    }
```

```rust
// src/tui/mod.rs
pub mod wizard;
pub mod widgets;

use std::io;
use crossterm::event::{self, Event, KeyCode, KeyEventKind};
use crossterm::terminal::{disable_raw_mode, enable_raw_mode, EnterAlternateScreen, LeaveAlternateScreen};
use crossterm::execute;
use ratatui::prelude::*;
use ratatui::widgets::{Block, Borders, List, ListItem, Paragraph, Clear};
use ratatui::Terminal;

use crate::config::AppConfig;
use crate::crypto;
use crate::crypto::keychain::{KeyStore, OsKeyStore};
use crate::error::{SubSwapError, Result};
use crate::guard::{CodexGuard, OsGuard};
use crate::paths::Paths;
use crate::profile::store::ProfileStore;
use crate::profile::switch;
use widgets::{Action, AppScreen, AppState};

pub fn run_tui(paths: &Paths) -> Result<()> {
    enable_raw_mode().map_err(|e| SubSwapError::Io(io::Error::new(io::ErrorKind::Other, e)))?;
    let mut stdout = io::stdout();
    execute!(stdout, EnterAlternateScreen)
        .map_err(|e| SubSwapError::Io(io::Error::new(io::ErrorKind::Other, e)))?;

    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend)
        .map_err(|e| SubSwapError::Io(io::Error::new(io::ErrorKind::Other, e)))?;

    let result = run_app(&mut terminal, paths);

    disable_raw_mode().ok();
    execute!(terminal.backend_mut(), LeaveAlternateScreen).ok();
    terminal.show_cursor().ok();

    result
}

fn run_app<B: Backend>(terminal: &mut Terminal<B>, paths: &Paths) -> Result<()> {
    let store = ProfileStore::load(paths)?;
    let mut state = AppState::from_index(&store.index);

    loop {
        terminal.draw(|f| render(f, &state, &store))
            .map_err(|e| SubSwapError::Io(io::Error::new(io::ErrorKind::Other, e)))?;

        if let Event::Key(key) = event::read()
            .map_err(|e| SubSwapError::Io(io::Error::new(io::ErrorKind::Other, e)))?
        {
            if key.kind != KeyEventKind::Press {
                continue;
            }

            match state.screen {
                AppScreen::Main => handle_main_key(&mut state, key.code, paths)?,
                AppScreen::ConfirmSwitch | AppScreen::ForceSwitch => {
                    handle_confirm_switch(&mut state, key.code, paths)?
                }
                AppScreen::ConfirmDelete => handle_confirm_delete(&mut state, key.code, paths)?,
                AppScreen::InputName => handle_input_name(&mut state, key.code, paths)?,
                AppScreen::InputNote => handle_input_note(&mut state, key.code, paths)?,
                AppScreen::ViewDecrypt => {
                    // Any key returns to main
                    state.screen = AppScreen::Main;
                    state.decrypt_output = None;
                }
            }
        }

        if state.should_quit {
            return Ok(());
        }
    }
}

fn handle_main_key(state: &mut AppState, key: KeyCode, _paths: &Paths) -> Result<()> {
    match key {
        KeyCode::Char('q') | KeyCode::Esc => state.should_quit = true,
        KeyCode::Up | KeyCode::Char('k') => state.move_up(),
        KeyCode::Down | KeyCode::Char('j') => state.move_down(),
        KeyCode::Enter => {
            state.pending_action = Some(Action::Switch);
            state.screen = AppScreen::ConfirmSwitch;
            state.message = None;
        }
        KeyCode::Char('a') => {
            state.pending_action = Some(Action::Add);
            state.screen = AppScreen::InputName;
            state.input_buffer.clear();
            state.message = Some("Enter new profile name:".into());
        }
        KeyCode::Char('r') => {
            state.pending_action = Some(Action::Rename);
            state.screen = AppScreen::InputName;
            state.input_buffer.clear();
            state.message = Some("Enter new name:".into());
        }
        KeyCode::Char('d') => {
            state.pending_action = Some(Action::Delete);
            state.screen = AppScreen::ConfirmDelete;
        }
        KeyCode::Char('n') => {
            state.pending_action = Some(Action::Note);
            state.screen = AppScreen::InputNote;
            state.input_buffer.clear();
            state.message = Some("Enter note:".into());
        }
        KeyCode::Char('v') => {
            state.pending_action = Some(Action::View);
            handle_view(state, _paths)?;
        }
        _ => {}
    }
    Ok(())
}

fn handle_confirm_switch(state: &mut AppState, key: KeyCode, paths: &Paths) -> Result<()> {
    match key {
        KeyCode::Char('y') | KeyCode::Enter => {
            if let Some(name) = state.selected_name().map(String::from) {
                if state.active_profile.as_deref() == Some(&name) {
                    state.message = Some(format!("'{name}' is already active."));
                    state.screen = AppScreen::Main;
                    return Ok(());
                }

                // Check process guard unless we're in ForceSwitch mode
                if state.screen != AppScreen::ForceSwitch {
                    let guard = OsGuard::new();
                    if let Err(SubSwapError::CodexRunning(_)) = guard.check() {
                        state.message = Some("Codex is running. Force switch? (y/n)".into());
                        state.screen = AppScreen::ForceSwitch;
                        return Ok(());
                    }
                }

                let config = AppConfig::load(paths)?;
                let keystore = OsKeyStore::new();
                let key = if config.encryption_enabled {
                    keystore.get_key()?
                } else {
                    [0u8; 32]
                };

                switch::switch_profile(paths, &name, &key, config.encryption_enabled)?;

                // Refresh state
                let store = ProfileStore::load(paths)?;
                *state = AppState::from_index(&store.index);
                state.message = Some(format!("Switched to '{name}'."));
            }
        }
        KeyCode::Char('n') | KeyCode::Esc => {
            state.screen = AppScreen::Main;
            state.message = None;
        }
        _ => {}
    }
    Ok(())
}

fn handle_confirm_delete(state: &mut AppState, key: KeyCode, paths: &Paths) -> Result<()> {
    match key {
        KeyCode::Char('y') => {
            if let Some(name) = state.selected_name().map(String::from) {
                let mut store = ProfileStore::load(paths)?;
                match store.index.remove(&name) {
                    Ok(_) => {
                        ProfileStore::delete_profile_dir(paths, &name)?;
                        store.save(paths)?;
                        *state = AppState::from_index(&store.index);
                        state.message = Some(format!("'{name}' removed."));
                    }
                    Err(e) => {
                        state.message = Some(format!("{e}"));
                        state.screen = AppScreen::Main;
                    }
                }
            }
        }
        KeyCode::Char('n') | KeyCode::Esc => {
            state.screen = AppScreen::Main;
            state.message = None;
        }
        _ => {}
    }
    Ok(())
}

fn handle_input_name(state: &mut AppState, key: KeyCode, paths: &Paths) -> Result<()> {
    match key {
        KeyCode::Enter => {
            let input = state.input_buffer.trim().to_string();
            if input.is_empty() {
                state.screen = AppScreen::Main;
                state.message = None;
                return Ok(());
            }

            match state.pending_action {
                Some(Action::Add) => {
                    let config = AppConfig::load(paths)?;
                    let keystore = OsKeyStore::new();
                    let key = if config.encryption_enabled {
                        keystore.get_key()?
                    } else {
                        [0u8; 32]
                    };
                    let mut store = ProfileStore::load(paths)?;
                    match switch::add_profile_from_codex(
                        paths, &mut store, &input, None, &key, config.encryption_enabled,
                    ) {
                        Ok(_) => {
                            *state = AppState::from_index(&store.index);
                            state.message = Some(format!("'{input}' added."));
                        }
                        Err(e) => {
                            state.message = Some(format!("{e}"));
                            state.screen = AppScreen::Main;
                        }
                    }
                }
                Some(Action::Rename) => {
                    if let Some(old) = state.selected_name().map(String::from) {
                        let mut store = ProfileStore::load(paths)?;
                        match store.index.rename(&old, &input) {
                            Ok(_) => {
                                ProfileStore::rename_profile_dir(paths, &old, &input)?;
                                store.save(paths)?;
                                *state = AppState::from_index(&store.index);
                                state.message = Some(format!("'{old}' renamed to '{input}'."));
                            }
                            Err(e) => {
                                state.message = Some(format!("{e}"));
                                state.screen = AppScreen::Main;
                            }
                        }
                    }
                }
                _ => {
                    state.screen = AppScreen::Main;
                }
            }
        }
        KeyCode::Esc => {
            state.screen = AppScreen::Main;
            state.message = None;
        }
        KeyCode::Backspace => {
            state.input_buffer.pop();
        }
        KeyCode::Char(c) => {
            state.input_buffer.push(c);
        }
        _ => {}
    }
    Ok(())
}

fn handle_input_note(state: &mut AppState, key: KeyCode, paths: &Paths) -> Result<()> {
    match key {
        KeyCode::Enter => {
            let text = state.input_buffer.trim().to_string();
            if let Some(name) = state.selected_name().map(String::from) {
                let mut store = ProfileStore::load(paths)?;
                store.index.set_note(&name, &text)?;
                store.save(paths)?;
                *state = AppState::from_index(&store.index);
                state.message = Some(format!("Note updated for '{name}'."));
            }
        }
        KeyCode::Esc => {
            state.screen = AppScreen::Main;
            state.message = None;
        }
        KeyCode::Backspace => {
            state.input_buffer.pop();
        }
        KeyCode::Char(c) => {
            state.input_buffer.push(c);
        }
        _ => {}
    }
    Ok(())
}

fn handle_view(state: &mut AppState, paths: &Paths) -> Result<()> {
    if let Some(name) = state.selected_name().map(String::from) {
        let config = AppConfig::load(paths)?;
        let keystore = OsKeyStore::new();
        let key = if config.encryption_enabled {
            keystore.get_key()?
        } else {
            [0u8; 32]
        };

        match switch::decrypt_profile_to_stdout(paths, &name, &key) {
            Ok((auth, config_toml)) => {
                state.decrypt_output = Some(format!(
                    "=== auth.json ===\n{auth}\n\n=== config.toml ===\n{config_toml}"
                ));
                state.screen = AppScreen::ViewDecrypt;
            }
            Err(e) => {
                state.message = Some(format!("{e}"));
            }
        }
    }
    Ok(())
}

fn render(f: &mut Frame, state: &AppState, store: &ProfileStore) {
    let area = f.area();

    match state.screen {
        AppScreen::ViewDecrypt => {
            let text = state.decrypt_output.as_deref().unwrap_or("(empty)");
            let para = Paragraph::new(text)
                .block(Block::default().borders(Borders::ALL).title(" Decrypted Profile (press any key to return) "));
            f.render_widget(para, area);
            return;
        }
        _ => {}
    }

    let chunks = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Min(5),
            Constraint::Length(3),
            Constraint::Length(2),
        ])
        .split(area);

    // Profile list
    let items: Vec<ListItem> = state
        .profile_names
        .iter()
        .enumerate()
        .map(|(i, name)| {
            let is_active = state.active_profile.as_deref() == Some(name.as_str());
            let marker = if is_active { "*" } else { " " };
            let notes = store
                .index
                .get(name)
                .and_then(|p| p.notes.as_deref())
                .unwrap_or("");
            let prefix = if i == state.selected { "> " } else { "  " };
            let line = format!("{prefix}{marker} {name:<16} {notes}");

            let style = if i == state.selected {
                Style::default().fg(Color::Yellow).add_modifier(Modifier::BOLD)
            } else if is_active {
                Style::default().fg(Color::Green)
            } else {
                Style::default()
            };

            ListItem::new(line).style(style)
        })
        .collect();

    let list = List::new(items)
        .block(Block::default().borders(Borders::ALL).title(" sub-swap "));
    f.render_widget(list, chunks[0]);

    // Key hints
    let hints = Paragraph::new(" [Enter] Switch  [a] Add  [r] Rename  [d] Delete  [n] Note  [v] View  [q] Quit")
        .block(Block::default().borders(Borders::ALL));
    f.render_widget(hints, chunks[1]);

    // Status / message / input
    let status_text = match state.screen {
        AppScreen::ConfirmSwitch => "Switch to this profile? (y/n)".to_string(),
        AppScreen::ForceSwitch => state.message.clone().unwrap_or_default(),
        AppScreen::ConfirmDelete => format!(
            "Delete '{}'? This cannot be undone. (y/n)",
            state.selected_name().unwrap_or("?")
        ),
        AppScreen::InputName | AppScreen::InputNote => {
            let prompt = state.message.as_deref().unwrap_or("Input:");
            format!("{prompt} {}_", state.input_buffer)
        }
        _ => state.message.clone().unwrap_or_default(),
    };

    let status = Paragraph::new(status_text);
    f.render_widget(status, chunks[2]);
}
```
